  var body = document.body;
  body.addEventListener('keydown', goJump, false);
  body.addEventListener('keyup', clearJump, false);
  function clearJump(e) {
      switch (e.keyCode) {
        case 49:
          document.querySelector('.idMush1').classList.remove('bounce-5');
          break;
        case 50:
          document.querySelector('.idMush2').classList.remove('bounce-5');
          break;
        case 51:
          document.querySelector('.idMush3').classList.remove('bounce-5');
          break;
        default:
          break;
      }
    }
  function goJump(e){
    switch (e.keyCode){
      case 49:
        document.querySelector('.idMush1').classList.add('bounce-5');
        break;
      case 50:
        document.querySelector('.idMush2').classList.add('bounce-5');
        break;
      case 51:
        document.querySelector('.idMush3').classList.add('bounce-5');
        break;
      default:
        break;
    }
  }
